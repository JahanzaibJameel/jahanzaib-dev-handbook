System Design Fundamentals

System design is the process of defining the architecture, components, modules, interfaces, and data for a system to satisfy specified requirements. It’s crucial for building scalable, maintainable, and robust applications.

🎯 What is System Design?

System design involves making high-level decisions about the organization and structure of software systems. It bridges the gap between requirements and implementation.

Why System Design Matters

  • Scalability: Handle growing user base and data
  • Reliability: Ensure system availability and fault tolerance
  • Performance: Meet response time and throughput requirements
  • Maintainability: Easy to modify and extend
  • Cost Efficiency: Optimize resource utilization

🏗️ System Design Fundamentals

Key Design Principles

// System design principles with examples
const designPrinciples = {
  // Single Responsibility Principle
  singleResponsibility: {
    description: "Each component should have one reason to change",
    example: `
// Bad: User service handles authentication, profile, and notifications
class UserService {
  authenticate() { /* ... */ }
  updateProfile() { /* ... */ }
  sendNotification() { /* ... */ }
}

// Good: Separate services for each responsibility
class AuthService { authenticate() { /* ... */ } }
class ProfileService { updateProfile() { /* ... */ } }
class NotificationService { sendNotification() { /* ... */ }
    `
  },
  
  // Separation of Concerns
  separationOfConcerns: {
    description: "Separate business logic from infrastructure concerns",
    example: `
// Separate layers for different concerns
class UserController {
  constructor(userService, emailService) {
    this.userService = userService;
    this.emailService = emailService;
  }
  
  async createUser(userData) {
    // Business logic
    const user = await this.userService.create(userData);
    
    // Infrastructure concern (notification)
    await this.emailService.sendWelcomeEmail(user.email);
    
    return user;
  }
}
    `
  },
  
  // Don't Repeat Yourself (DRY)
  dry: {
    description: "Avoid code duplication through abstraction",
    example: `
// Bad: Duplicate validation logic
function createUser(userData) {
  if (!userData.email || !userData.email.includes('@')) {
    throw new Error('Invalid email');
  }
  // ... create user
}

function updateUser(userId, userData) {
  if (!userData.email || !userData.email.includes('@')) {
    throw new Error('Invalid email');
  }
  // ... update user
}

// Good: Extract validation to reusable function
function validateEmail(email) {
  return email && email.includes('@');
}

function createUser(userData) {
  if (!validateEmail(userData.email)) {
    throw new Error('Invalid email');
  }
  // ... create user
}
    `
  }
};

Load Balancing Strategies

// Load balancing algorithms
const loadBalancingAlgorithms = {
  roundRobin: {
    description: "Distribute requests sequentially across servers",
    implementation: `
class RoundRobinLoadBalancer {
  constructor(servers) {
    this.servers = servers;
    this.currentIndex = 0;
  }
  
  getNextServer() {
    const server = this.servers[this.currentIndex];
    this.currentIndex = (this.currentIndex + 1) % this.servers.length;
    return server;
  }
}
    `,
    pros: ["Simple", "Even distribution"],
    cons: ["Ignores server capacity", "No health checks"]
  },
  
  leastConnections: {
    description: "Route to server with fewest active connections",
    implementation: `
class LeastConnectionsLoadBalancer {
  constructor(servers) {
    this.servers = servers.map(server => ({
      ...server,
      connections: 0
    }));
  }
  
  getNextServer() {
    return this.servers.reduce((min, current) => 
      current.connections < min.connections ? current : min
    );
  }
  
  incrementConnections(server) {
    server.connections++;
  }
  
  decrementConnections(server) {
    server.connections--;
  }
}
    `,
    pros: ["Considers server load", "Better for long connections"],
    cons: ["More complex", "Requires connection tracking"]
  }
};

🚀 Caching Strategies

Cache Patterns

// Caching patterns with implementations
const cachingPatterns = {
  cacheAside: {
    description: "Application manages cache explicitly",
    implementation: `
class CacheAsideService {
  constructor(database, cache) {
    this.db = database;
    this.cache = cache;
  }
  
  async get(key) {
    // Try cache first
    let data = await this.cache.get(key);
    
    if (!data) {
      // Cache miss - get from database
      data = await this.db.get(key);
      
      if (data) {
        // Populate cache
        await this.cache.set(key, data, { ttl: 3600 });
      }
    }
    
    return data;
  }
  
  async set(key, data) {
    // Update both cache and database
    await Promise.all([
      this.db.set(key, data),
      this.cache.set(key, data, { ttl: 3600 })
    ]);
  }
}
    `,
    pros: ["Flexible", "Cache failures don't break app"],
    cons: ["Stale data possible", "More complex code"]
  },
  
  readThrough: {
    description: "Cache manages database reads",
    implementation: `
class ReadThroughCache {
  constructor(database, cache) {
    this.db = database;
    this.cache = cache;
  }
  
  async get(key) {
    // Cache handles database lookup
    return await this.cache.getOrLoad(key, async () => {
      return await this.db.get(key);
    });
  }
}
    `,
    pros: ["Simpler application code", "Always consistent"],
    cons: ["Cache becomes bottleneck", "Less flexible"]
  },
  
  writeThrough: {
    description: "Write to both cache and database synchronously",
    implementation: `
class WriteThroughCache {
  constructor(database, cache) {
    this.db = database;
    this.cache = cache;
  }
  
  async set(key, data) {
    // Write to both synchronously
    await Promise.all([
      this.db.set(key, data),
      this.cache.set(key, data, { ttl: 3600 })
    ]);
  }
}
    `,
    pros: ["Data consistency", "Read performance"],
    cons: ["Write latency", "Cache size management"]
  }
};

Cache Invalidation Strategies

// Cache invalidation patterns
const cacheInvalidation = {
  ttl: {
    description: "Time-based expiration",
    implementation: `
// Set TTL when writing to cache
await cache.set(key, data, { ttl: 3600 }); // 1 hour
    `,
    pros: ["Simple", "No invalidation logic needed"],
    cons: ["May serve stale data", "Fixed expiration regardless of updates"]
  },
  
  writeInvalidate: {
    description: "Invalidate cache on data changes",
    implementation: `
class WriteInvalidateCache {
  async updateUser(userId, userData) {
    // Update database
    await this.db.update(userId, userData);
    
    // Invalidate cache
    await this.cache.delete(\`user:\${userId}\`);
  }
}
    `,
    pros: ["Always fresh data", "Immediate consistency"],
    cons: ["More complex", "Cache stampede possible"]
  },
  
  lazyLoading: {
    description: "Invalidate on next access",
    implementation: `
class LazyLoadingCache {
  async get(key) {
    let data = await this.cache.get(key);
    
    if (data && data.isStale()) {
      // Remove stale data
      await this.cache.delete(key);
      data = null;
    }
    
    if (!data) {
      // Load fresh data
      data = await this.db.get(key);
      await this.cache.set(key, data);
    }
    
    return data;
  }
}
    `,
    pros: ["No wasted cache updates", "Simple"],
    cons: ["First request after update is slow", "Stale data possible"]
  }
};

🔄 Message Queues & Event Systems

Message Queue Patterns

// Message queue patterns for system design
const messageQueuePatterns = {
  pointToPoint: {
    description: "One producer sends to one consumer",
    useCase: "Task processing, command handling",
    implementation: `
class PointToPointQueue {
  constructor() {
    this.queue = [];
    this.processing = false;
  }
  
  async send(message) {
    this.queue.push(message);
    this.process();
  }
  
  async process() {
    if (this.processing || this.queue.length === 0) return;
    
    this.processing = true;
    
    while (this.queue.length > 0) {
      const message = this.queue.shift();
      await this.handleMessage(message);
    }
    
    this.processing = false;
  }
}
    `
  },
  
  publishSubscribe: {
    description: "One publisher sends to multiple subscribers",
    useCase: "Event notifications, data distribution",
    implementation: `
class PubSubSystem {
  constructor() {
    this.subscribers = new Map();
  }
  
  subscribe(topic, callback) {
    if (!this.subscribers.has(topic)) {
      this.subscribers.set(topic, []);
    }
    this.subscribers.get(topic).push(callback);
  }
  
  publish(topic, message) {
    const callbacks = this.subscribers.get(topic) || [];
    callbacks.forEach(callback => callback(message));
  }
}
    `
  }
};

    description: "Prevent abuse and attacks",
    implementation: `
class RateLimiter {
  constructor(maxRequests, windowMs) {
    this.maxRequests = maxRequests;
    this.windowMs = windowMs;
    this.requests = new Map();
  }
  
  isAllowed(clientId) {
    const now = Date.now();
    const windowStart = now - this.windowMs;
    
    if (!this.requests.has(clientId)) {
      this.requests.set(clientId, []);
    }
    
    const clientRequests = this.requests.get(clientId);
    
    // Remove old requests
    const validRequests = clientRequests.filter(time => time > windowStart);
    this.requests.set(clientId, validRequests);
    
    if (validRequests.length >= this.maxRequests) {
      return false;
    }
    
    validRequests.push(now);
    return true;
  }
}
    `
  }
