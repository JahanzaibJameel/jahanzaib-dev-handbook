<h1 id="components--props-in-react-native"><a class="header" href="#components--props-in-react-native">Components &amp; Props in React Native</a></h1> <p>Components are the building blocks of React Native applications. They encapsulate reusable pieces of UI and logic.</p> <h2 id="-understanding-components"><a class="header" href="#-understanding-components">🧩 Understanding Components</a></h2> <h3 id="what-is-a-component"><a class="header" href="#what-is-a-component">What is a Component?</a></h3> <p>A component is a self-contained, reusable piece of UI that can accept input (props) and manage its own state.</p> <pre><code class="language-javascript">import React from 'react'; import { View, Text, StyleSheet } from 'react-native'; // Functional Component const WelcomeCard = () =&gt; { return ( &lt;View style={styles.container}&gt; &lt;Text style={styles.title}&gt;Welcome!&lt;/Text&gt; &lt;/View&gt; ); }; const styles = StyleSheet.create({ container: { padding: 20, backgroundColor: '#f0f0f0', borderRadius: 8, }, title: { fontSize: 18, fontWeight: 'bold', }, }); export default WelcomeCard; </code></pre> <h2 id="-props-properties"><a class="header" href="#-props-properties">📦 Props (Properties)</a></h2> <p>Props are read-only attributes that pass data from parent to child components.</p> <h3 id="basic-props-usage"><a class="header" href="#basic-props-usage">Basic Props Usage</a></h3> <pre><code class="language-javascript">import React from 'react'; import { View, Text, StyleSheet } from 'react-native'; const UserCard = ({ name, email, avatar }) =&gt; { return ( &lt;View style={styles.card}&gt; &lt;Text style={styles.name}&gt;{name}&lt;/Text&gt; &lt;Text style={styles.email}&gt;{email}&lt;/Text&gt; &lt;/View&gt; ); }; // Parent Component const UserList = () =&gt; { const users = [ { name: 'John Doe', email: 'john@example.com' }, { name: 'Jane Smith', email: 'jane@example.com' }, ]; return ( &lt;View&gt; {users.map((user, index) =&gt; ( &lt;UserCard key={index} name={user.name} email={user.email} /&gt; ))} &lt;/View&gt; ); }; </code></pre> <h3 id="prop-types-and-validation"><a class="header" href="#prop-types-and-validation">Prop Types and Validation</a></h3> <pre><code class="language-javascript">import React from 'react'; import { View, Text, StyleSheet } from 'react-native'; import PropTypes from 'prop-types'; const Button = ({ title, onPress, disabled, color }) =&gt; { return ( &lt;TouchableOpacity style={[styles.button, { backgroundColor: color }]} onPress={onPress} disabled={disabled} &gt; &lt;Text style={styles.buttonText}&gt;{title}&lt;/Text&gt; &lt;/TouchableOpacity&gt; ); }; Button.propTypes = { title: PropTypes.string.isRequired, onPress: PropTypes.func.isRequired, disabled: PropTypes.bool, color: PropTypes.string, }; Button.defaultProps = { disabled: false, color: '#007AFF', }; </code></pre> <h2 id="-core-components"><a class="header" href="#-core-components">🎨 Core Components</a></h2> <h3 id="view"><a class="header" href="#view">View</a></h3> <p>The most fundamental component for building UI:</p> <pre><code class="language-javascript">import { View, StyleSheet } from 'react-native'; const Container = () =&gt; { return ( &lt;View style={styles.container}&gt; &lt;View style={styles.header} /&gt; &lt;View style={styles.content} /&gt; &lt;View style={styles.footer} /&gt; &lt;/View&gt; ); }; const styles = StyleSheet.create({ container: { flex: 1, }, header: { height: 60, backgroundColor: '#fff', }, content: { flex: 1, backgroundColor: '#f5f5f5', }, footer: { height: 50, backgroundColor: '#fff', }, }); </code></pre> <h3 id="text"><a class="header" href="#text">Text</a></h3> <p>For displaying text content:</p> <pre><code class="language-javascript">import { Text, StyleSheet } from 'react-native'; const TextExamples = () =&gt; { return ( &lt;View&gt; &lt;Text style={styles.title}&gt;Main Title&lt;/Text&gt; &lt;Text style={styles.subtitle}&gt;Subtitle Text&lt;/Text&gt; &lt;Text style={styles.body}&gt;Regular body text content&lt;/Text&gt; &lt;Text style={styles.caption}&gt;Small caption text&lt;/Text&gt; &lt;/View&gt; ); }; const styles = StyleSheet.create({ title: { fontSize: 24, fontWeight: 'bold', color: '#000', }, subtitle: { fontSize: 18, fontWeight: '600', color: '#333', }, body: { fontSize: 16, color: '#666', lineHeight: 24, }, caption: { fontSize: 12, color: '#999', }, }); </code></pre> <h3 id="image"><a class="header" href="#image">Image</a></h3> <p>For displaying images:</p> <pre><code class="language-javascript">import { Image, StyleSheet, View } from 'react-native'; const ImageExamples = () =&gt; { return ( &lt;View&gt; {/* Local Image */} &lt;Image source={require('./assets/logo.png')} style={styles.localImage} /&gt; {/* Network Image */} &lt;Image source={{ uri: 'https://example.com/image.jpg' }} style={styles.networkImage} /&gt; {/* Image with Placeholder */} &lt;Image source={require('./assets/profile.jpg')} style={styles.avatar} defaultSource={require('./assets/placeholder.png')} /&gt; &lt;/View&gt; ); }; </code></pre> <h3 id="touchableopacity"><a class="header" href="#touchableopacity">TouchableOpacity</a></h3> <p>For creating pressable elements:</p> <pre><code class="language-javascript">import { TouchableOpacity, Text, StyleSheet } from 'react-native'; const CustomButton = ({ title, onPress, variant }) =&gt; { const buttonStyle = variant === 'primary' ? styles.primaryButton : styles.secondaryButton; return ( &lt;TouchableOpacity style={buttonStyle} onPress={onPress}&gt; &lt;Text style={styles.buttonText}&gt;{title}&lt;/Text&gt; &lt;/TouchableOpacity&gt; ); }; </code></pre> <h2 id="-custom-components"><a class="header" href="#-custom-components">🏗️ Custom Components</a></h2> <h3 id="creating-reusable-components"><a class="header" href="#creating-reusable-components">Creating Reusable Components</a></h3> <pre><code class="language-javascript">import React from 'react'; import { View, Text, TouchableOpacity, StyleSheet } from 'react-native'; const Card = ({ children, title, onPress }) =&gt; { return ( &lt;TouchableOpacity style={styles.card} onPress={onPress}&gt; {title &amp;&amp; &lt;Text style={styles.title}&gt;{title}&lt;/Text&gt;} &lt;View style={styles.content}&gt; {children} &lt;/View&gt; &lt;/TouchableOpacity&gt; ); }; const InputField = ({ label, value, onChangeText, placeholder, secure }) =&gt; { return ( &lt;View style={styles.inputContainer}&gt; &lt;Text style={styles.label}&gt;{label}&lt;/Text&gt; &lt;TextInput style={styles.input} value={value} onChangeText={onChangeText} placeholder={placeholder} secureTextEntry={secure} /&gt; &lt;/View&gt; ); }; const LoadingSpinner = ({ size, color }) =&gt; { return ( &lt;ActivityIndicator size={size || 'large'} color={color || '#007AFF'} /&gt; ); }; </code></pre> <h2 id="-component-composition"><a class="header" href="#-component-composition">🔄 Component Composition</a></h2> <h3 id="building-complex-uis"><a class="header" href="#building-complex-uis">Building Complex UIs</a></h3> <pre><code class="language-javascript">import React from 'react'; import { View, ScrollView, StyleSheet } from 'react-native'; import Card from './Card'; import UserCard from './UserCard'; import LoadingSpinner from './LoadingSpinner'; const UserFeed = ({ users, loading, onUserPress }) =&gt; { if (loading) { return &lt;LoadingSpinner /&gt;; } return ( &lt;ScrollView style={styles.container}&gt; {users.map(user =&gt; ( &lt;Card key={user.id} onPress={() =&gt; onUserPress(user)}&gt; &lt;UserCard name={user.name} email={user.email} avatar={user.avatar} /&gt; &lt;/Card&gt; ))} &lt;/ScrollView&gt; ); }; </code></pre> <h2 id="-best-practices"><a class="header" href="#-best-practices">🎯 Best Practices</a></h2> <h3 id="1-single-responsibility"><a class="header" href="#1-single-responsibility">1. Single Responsibility</a></h3> <p>Each component should have one clear purpose:</p> <pre><code class="language-javascript">// Good: Single responsibility const Avatar = ({ source, size }) =&gt; ( &lt;Image source={source} style={[styles.avatar, { width: size, height: size }]} /&gt; ); // Bad: Multiple responsibilities const UserSection = ({ user, onEdit, onDelete, onMessage }) =&gt; { // Too many responsibilities in one component }; </code></pre> <h3 id="2-props-destructuring"><a class="header" href="#2-props-destructuring">2. Props Destructuring</a></h3> <p>Always destructure props for cleaner code:</p> <pre><code class="language-javascript">// Good const UserCard = ({ name, email, avatar }) =&gt; { return ( &lt;View&gt; &lt;Text&gt;{name}&lt;/Text&gt; &lt;Text&gt;{email}&lt;/Text&gt; &lt;/View&gt; ); }; // Avoid const UserCard = (props) =&gt; { return ( &lt;View&gt; &lt;Text&gt;{props.name}&lt;/Text&gt; &lt;Text&gt;{props.email}&lt;/Text&gt; &lt;/View&gt; ); }; </code></pre> <h3 id="3-default-props"><a class="header" href="#3-default-props">3. Default Props</a></h3> <p>Always provide sensible defaults:</p> <pre><code class="language-javascript">const Button = ({ title, onPress, disabled = false, variant = 'primary', size = 'medium' }) =&gt; { // Component implementation }; </code></pre> <h3 id="4-component-naming"><a class="header" href="#4-component-naming">4. Component Naming</a></h3> <p>Use descriptive, PascalCase names:</p> <pre><code class="language-javascript">// Good const UserProfileCard = () =&gt; {}; const NavigationHeader = () =&gt; {}; const SearchInputField = () =&gt; {}; // Avoid const Card = () =&gt; {}; // Too generic const Header = () =&gt; {}; // Too generic const Input = () =&gt; {}; // Too generic </code></pre> <h2 id="-testing-components"><a class="header" href="#-testing-components">🧪 Testing Components</a></h2> <h3 id="basic-component-testing"><a class="header" href="#basic-component-testing">Basic Component Testing</a></h3> <pre><code class="language-javascript">import React from 'react'; import { render, fireEvent } from '@testing-library/react-native'; import Button from '../Button'; describe('Button Component', () =&gt; { it('renders correctly with title', () =&gt; { const { getByText } = render( &lt;Button title="Press me" onPress={() =&gt; {}} /&gt; ); expect(getByText('Press me')).toBeTruthy(); }); it('calls onPress when pressed', () =&gt; { const mockOnPress = jest.fn(); const { getByText } = render( &lt;Button title="Press me" onPress={mockOnPress} /&gt; ); fireEvent.press(getByText('Press me')); expect(mockOnPress).toHaveBeenCalled(); }); }); </code></pre> <h2 id="-component-library"><a class="header" href="#-component-library">📚 Component Library</a></h2> <h3 id="building-a-component-library"><a class="header" href="#building-a-component-library">Building a Component Library</a></h3> <pre><code class="language-javascript">// components/index.js export { default as Button } from './Button'; export { default as Card } from './Card'; export { default as Input } from './Input'; export { default as Avatar } from './Avatar'; // Usage in other files import { Button, Card, Input } from '../components'; </code></pre> <h3 id="theme-integration"><a class="header" href="#theme-integration">Theme Integration</a></h3> <pre><code class="language-javascript">// theme.js export const colors = { primary: '#007AFF', secondary: '#5856D6', success: '#34C759', warning: '#FF9500', error: '#FF3B30', background: '#F2F2F7', surface: '#FFFFFF', text: '#000000', textSecondary: '#8E8E93', }; export const spacing = { xs: 4, sm: 8, md: 16, lg: 24, xl: 32, }; export const typography = { h1: { fontSize: 32, fontWeight: 'bold', }, h2: { fontSize: 24, fontWeight: 'bold', }, body: { fontSize: 16, fontWeight: 'normal', }, caption: { fontSize: 12, fontWeight: 'normal', }, }; // Themed Component const ThemedButton = ({ title, onPress }) =&gt; { return ( &lt;TouchableOpacity style={[styles.button, { backgroundColor: colors.primary }]} onPress={onPress} &gt; &lt;Text style={[styles.text, { color: colors.surface }]}&gt; {title} &lt;/Text&gt; &lt;/TouchableOpacity&gt; ); }; </code></pre> <hr> <p><strong>Next Up</strong>: Learn about navigation to connect your components into a complete app! 🚀</p>
